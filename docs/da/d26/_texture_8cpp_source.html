176 Texture* TextureHandler::binarySearchTextures(const std::string& key) {
177  unsigned int mid = 0;
178  unsigned int left = 0;
179  unsigned int right = textureLibrary.size();
180 
181  while (left < right) {
182  mid = left + (right - left) / 2;
183 
184  if (key > textureLibrary.at(mid)->getLocation()) {
185  left = mid + 1;
186  } else if (key < textureLibrary.at(mid)->getLocation()) {
187  right = mid;
188  } else {
189  return textureLibrary.at(mid);
190  }
